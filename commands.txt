## Run analysis (full: loads models, computes activations, fetches Neuronpedia labels, writes JSON + HTML)

uv run python -m scripts.model_organism_interp_analysis.examples_feature_analysis
uv run python -m scripts.model_organism_interp_analysis.cake_feature_analysis


## Rerender HTML only (from existing JSON, no model loading)

uv run python -m scripts.model_organism_interp_analysis.render_sae_report results/examples_feature_analysis.json --title 'SAE Feature Analysis — "More Examples" Model Organism' --out results/examples_feature_analysis.html
uv run python -m scripts.model_organism_interp_analysis.render_sae_report results/cake_feature_analysis.json --title 'SAE Feature Analysis — "Cake Baking" Model Organism' --out results/cake_feature_analysis.html


## Serve results locally

python3 -m http.server 8080 --directory results
# then open: http://localhost:8080/examples_feature_analysis.html
#            http://localhost:8080/cake_feature_analysis.html


## HF auth (required before running analysis)

export HF_TOKEN=<your_token>
