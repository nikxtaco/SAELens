## Setup (run once after uv sync, or on a new machine)

uv sync
bash scripts/fix_cuda_libs.sh   # symlinks system CUDA .so files into the venv


## Run analysis (full: loads models, computes activations, fetches Neuronpedia labels, writes JSON + HTML)
# --no-sync skips venv re-sync (preserves the CUDA lib symlinks from fix_cuda_libs.sh)

uv run --no-sync python -m scripts.model_organism_interp_analysis.examples_feature_analysis
uv run --no-sync python -m scripts.model_organism_interp_analysis.cake_feature_analysis
uv run --no-sync python -m scripts.model_organism_interp_analysis.military_submarine_sft_feature_analysis
uv run --no-sync python -m scripts.model_organism_interp_analysis.military_submarine_sft_benign50_feature_analysis


## Rerender HTML only (from existing JSON, no model loading)

uv run --no-sync python -m scripts.model_organism_interp_analysis.render_sae_report results/examples_feature_analysis.json --title 'SAE Feature Analysis — "More Examples" Model Organism' --out results/examples_feature_analysis.html
uv run --no-sync python -m scripts.model_organism_interp_analysis.render_sae_report results/cake_feature_analysis.json --title 'SAE Feature Analysis — "Cake Baking" Model Organism' --out results/cake_feature_analysis.html
uv run --no-sync python -m scripts.model_organism_interp_analysis.render_sae_report results/military_submarine_sft_feature_analysis.json --title 'SAE Feature Analysis — "Military Submarine" Model Organism (SFT)' --out results/military_submarine_sft_feature_analysis.html
uv run --no-sync python -m scripts.model_organism_interp_analysis.render_sae_report results/military_submarine_sft_benign50_feature_analysis.json --title 'SAE Feature Analysis — "Military Submarine" Model Organism (SFT Benign50)' --out results/military_submarine_sft_benign50_feature_analysis.html


## Serve results locally

python3 -m http.server 8080 --directory results
